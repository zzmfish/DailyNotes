    PostgreSQL

    🔖数据库 

    设置

    设置初始密码

    # 切换用户 postgres
sudo -i -u postgres
# 进入 shell
psql
# 设置密码
\password postgres
# 退出
\q
exit

    给当前登录用户分配权限

    sudo -u postgres createuser --superuser $USER
sudo -u postgres createdb $USER

    允许远程访问

    1、编辑文件：/etc/postgresql/13/main/postgresql.conf

    #listen_addresses = 'localhost'
listen_addresses = '*'

    2、编辑文件：/etc/postgresql/13/main/pg_hba.conf

    # IPv4 local connections:
#host    all             all             127.0.0.1/32            md5
host    all             all             0.0.0.0/0            md5

    重启防火墙和数据库

    sudo ufw allow 5432/tcp
sudo systemctl restart postgresql

    SQL 语句

    数据库

    -- 显示 db 占用的磁盘空间（GB）
SELECT pg_database_size(db_name) / 1024 / 1024 / 1024;

-- 显示所有 table 占用的磁盘空间（GB）
SELECT table_name,
    pg_table_size(quote_ident(table_name)) / 1024 / 1024 / 1024
FROM information_schema.tables
WHERE table_schema = 'public'
ORDER BY 2;

    表

    ALTER TABLE
    -- 增加列
ALTER TABLE table_name ADD column_name column_type;

-- 重命名列
ALTER TABLE table_name RENAME COLUMN column_name TO new_name;

-- 删除列
ALTER TABLE table_name DROP COLUMN column_name;

    索引

    CREATE INDEX
    -- 唯一索引
CREATE UNIQUE INDEX index_name
ON table_name (column_name);
